\paragraph{A: Problems on Reviewing of Orthogonal transformations, 
Rotations, Reflections and Rigid Motions
  in $R^n$.}
\begin{itemize}

{\item a) Let $\rho$ and $\tau$ be two orthogonal transformations on
  an Euclidean space $(V^n, <,>)$.  Prove that the composition of
  $\rho$ and $\tau$ is again an orthogonal transformation of $(V^n,
  <,>)$.  So is the inverse of $\rho$.}
 
{\item b) Problem 6 on page 23, Section 1-5, Baby Do Carmo.}
\end{itemize}
